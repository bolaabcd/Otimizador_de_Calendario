código SQL usado:

USE otimizador_db;

-- 1. Criando a tabela USER
CREATE TABLE USER (
    user_id INT PRIMARY KEY,
    user_name VARCHAR(255)
);

-- 2. Criando a tabela DATES
CREATE TABLE DATES (
    date_id INT PRIMARY KEY, /*Pensei em uma chave do tipo YYMMDDYYMMDD que merge start_date e end_date*/
    start_date DATE,
    end_date DATE
);

-- 3. Criando a tabela PLACES
CREATE TABLE PLACES (
    place_id INT PRIMARY KEY,
    address VARCHAR(255)
);

-- 4. Criando a tabela ACTIVITIES
CREATE TABLE ACTIVITIES (
    activity_id INT PRIMARY KEY,
    user_id INT,
    date_id INT,
    place_id INT,
    importance INT,
    FOREIGN KEY (date_id) REFERENCES DATES(date_id),
    FOREIGN KEY (place_id) REFERENCES PLACES(place_id),
    FOREIGN KEY (user_id) REFERENCES USER(user_id)
);
